import numpy as np import pandas as pd import tensorflow as tf from sklearn.preprocessing import StandardScaler from sklearn.model_selection import train_test_split from tensorflow.keras.models import Sequential from tensorflow.keras.layers import Dense, Activation, Dropout from tensorflow.keras.optimizers import Adam # –ó–∞–≥—Ä—É–∑–∫–∞ –∏ –ø–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–∞–Ω–Ω—ã—Ö dataset = pd.read_csv('dataset_prob.csv') X = dataset[['num_links', 'num_words', 'has_offer', 'sender_score', 'all_caps']].values Y = dataset['is_spam'].values # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö 60/20/20 X_train, X_test, Y_train, Y_test = train_test_split(X, Y, test_size=0.2, stratify=Y) X_train, X_val, Y_train, Y_val = train_test_split(X_train, Y_train, test_size=0.25, stratify=Y_train) # StandardScaler –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è scaler = StandardScaler() X_train = scaler.fit_transform(X_train) X_val = scaler.transform(X_val) X_test = scaler.transform(X_test) # –°–æ–∑–¥–∞–Ω–∏–µ –º–æ–¥–µ–ª–∏ —Å ReLU –∏ L2 —Ä–µ–≥—É–ª—è—Ä–∏–∑–∞—Ü–∏–µ–π model = Sequential() model.add(Dense(12, input_shape=(5,), kernel_regularizer=tf.keras.regularizers.l2(0.001))) model.add(Activation('relu')) model.add(Dropout(0.2)) model.add(Dense(6, kernel_regularizer=tf.keras.regularizers.l2(0.001))) model.add(Activation('relu')) model.add(Dropout(0.2)) model.add(Dense(2)) model.add(Activation('softmax')) # –ö–æ–º–ø–∏–ª—è—Ü–∏—è —Å Adam –æ–ø—Ç–∏–º–∏–∑–∞—Ç–æ—Ä–æ–º model.compile(loss='sparse_categorical_crossentropy', optimizer=Adam(learning_rate=0.0005), metrics=['accuracy']) # –û–±—É—á–µ–Ω–∏–µ model.fit(X_train, Y_train, batch_size=128, epochs=100, validation_data=(X_val, Y_val)) # –û—Ü–µ–Ω–∫–∞ test_loss, test_accuracy = model.evaluate(X_test, Y_test) print(f"Test Accuracy: {test_accuracy:.4f}")
